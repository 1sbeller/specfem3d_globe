28  subroutine create_serial_name_database(prname,iproc,iregion_code, &
29  local_path,nproctot,output_files)
30 
31 ! create name of the database for serial codes (AVS_DX and codes to check buffers)
32 
33  use constants
34 
35  implicit none
36 
37  integer iproc,iregion_code,NPROCTOT
38 
39 ! name of the database file
40  character(len=MAX_STRING_LEN) prname,procname,base_path,serial_prefix, &
41  LOCAL_PATH,OUTPUT_FILES
42 
43  integer iprocloop
44  integer, dimension(:), allocatable :: num_active_proc
45 
46 ! create the name for the database of the current slide and region
47  write(procname,"('/proc',i6.6,'_reg',i1,'_')") iproc,iregion_code
48 
49 ! on a machine with local disks, path on frontend can be different from local paths
50  if (.not. local_path_is_also_global) then
51 
52 ! allocate array for active processors
53  allocate(num_active_proc(0:nproctot-1))
54 
55 ! read filtered file with name of active machines
56  open(unit=iin,file=trim(output_files)//'/filtered_machines.txt',status='old',action='read')
57  do iprocloop = 0,nproctot-1
58  read(iin,*) num_active_proc(iprocloop)
59  enddo
60  close(iin)
61 
62 ! create the serial prefix pointing to the correct machine
63  write(serial_prefix,"('/auto/scratch_n',i6.6,'/')") num_active_proc(iproc)
64 
65 ! suppress everything until the last "/" to define the base name of local path
66 ! this is system dependent since it assumes the disks are mounted remotely
67  base_path = local_path(index(local_path,'/',.true.)+1:len_trim(local_path))
68 
69 ! create full name with path
70  prname = trim(serial_prefix) // trim(base_path) // procname
71 
72 ! deallocate array
73  deallocate(num_active_proc)
74 
75 ! on shared-memory machines, global path is the same as local path
76  else
77 
78 ! create full name with path
79  prname = trim(local_path) // procname
80 
81  endif
82 
83  end subroutine create_serial_name_database
