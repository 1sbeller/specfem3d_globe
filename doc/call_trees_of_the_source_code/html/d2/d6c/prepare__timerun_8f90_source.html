223  subroutine prepare_timerun_mass_matrices()
224 
225  use specfem_par
226  use specfem_par_crustmantle
227  use specfem_par_innercore
228  use specfem_par_outercore
229  implicit none
230 
231  if (myrank == 0) then
232  write(imain,*) "preparing mass matrices"
233  call flush_imain()
234  endif
235 
236  ! mass matrices
237  !
238  ! in the case of Stacey boundary conditions, add C*deltat/2 contribution to the mass matrix
239  ! on Stacey edges for the crust_mantle and outer_core regions but not for the inner_core region
240  ! thus the mass matrix must be replaced by three mass matrices including the "C" damping matrix
241  !
242  ! if absorbing_conditions are not set or if NCHUNKS=6, only one mass matrix is needed
243  ! for the sake of performance, only "rmassz" array will be filled and "rmassx" & "rmassy" will be obsolete
244 
245 
246  ! mass matrices need to be assembled with MPI here once and for all
247  call prepare_timerun_rmass_assembly()
248 
249  ! checks that all the mass matrices are positive
250  ! ocean load
251  if (oceans_val) then
252  if (minval(rmass_ocean_load) <= 0._custom_real) &
253  call exit_mpi(myrank,'negative mass matrix term for the oceans')
254  endif
255 
256  ! checks mass matrices
257 
258  ! crust/mantle
259  if (minval(rmassx_crust_mantle) <= 0._custom_real) &
260  call exit_mpi(myrank,'negative mass matrix term for the crust_mantle rmassx')
261  if (minval(rmassy_crust_mantle) <= 0._custom_real) &
262  call exit_mpi(myrank,'negative mass matrix term for the crust_mantle rmassy')
263  if (minval(rmassz_crust_mantle) <= 0._custom_real) &
264  call exit_mpi(myrank,'negative mass matrix term for the crust_mantle rmassz')
265  ! kernel simulations
266  if (simulation_type == 3) then
267  if (minval(b_rmassx_crust_mantle) <= 0._custom_real) &
268  call exit_mpi(myrank,'negative mass matrix term for the b_crust_mantle b_rmassx')
269  if (minval(b_rmassy_crust_mantle) <= 0._custom_real) &
270  call exit_mpi(myrank,'negative mass matrix term for the b_crust_mantle b_rmassy')
271  if (minval(b_rmassz_crust_mantle) <= 0._custom_real) &
272  call exit_mpi(myrank,'negative mass matrix term for the b_crust_mantle b_rmassz')
273  endif
274 
275  ! inner core
276  ! checks mass matrices for rotation
277  if (minval(rmassx_inner_core) <= 0._custom_real) &
278  call exit_mpi(myrank,'negative mass matrix term for the inner core rmassx')
279  if (minval(rmassy_inner_core) <= 0._custom_real) &
280  call exit_mpi(myrank,'negative mass matrix term for the inner core rmassy')
281  if (minval(rmassz_inner_core) <= 0._custom_real) &
282  call exit_mpi(myrank,'negative mass matrix term for the inner core rmassz')
283  ! kernel simulations
284  if (simulation_type == 3) then
285  if (minval(b_rmassx_inner_core) <= 0._custom_real) &
286  call exit_mpi(myrank,'negative mass matrix term for the b_rmassx_inner_core')
287  if (minval(b_rmassy_inner_core) <= 0._custom_real) &
288  call exit_mpi(myrank,'negative mass matrix term for the b_rmassy_inner_core')
289  if (minval(b_rmassz_inner_core) <= 0._custom_real) &
290  call exit_mpi(myrank,'negative mass matrix term for the b_rmassz_inner_core')
291  endif
292 
293  ! outer core
294  if (minval(rmass_outer_core) <= 0._custom_real) &
295  call exit_mpi(myrank,'negative mass matrix term for the outer core')
296  if (simulation_type == 3) then
297  if (minval(b_rmass_outer_core) <= 0._custom_real) &
298  call exit_mpi(myrank,'negative mass matrix term for the outer core b_rmass')
299  endif
300 
301  ! mass matrix inversions
302  ! for efficiency, invert final mass matrix once and for all on each slice
303  ! ocean load
304  if (oceans_val) rmass_ocean_load = 1._custom_real / rmass_ocean_load
305 
306  ! mass matrices on Stacey edges
307  ! crust/mantle
308  if (((nchunks_val /= 6 .and. absorbing_conditions) .or. &
309  (rotation_val .and. exact_mass_matrix_for_rotation_val))) then
310  rmassx_crust_mantle = 1._custom_real / rmassx_crust_mantle
311  rmassy_crust_mantle = 1._custom_real / rmassy_crust_mantle
312  endif
313  if (simulation_type == 3) then
314  if (rotation_val .and. exact_mass_matrix_for_rotation_val) then
315  b_rmassx_crust_mantle = 1._custom_real / b_rmassx_crust_mantle
316  b_rmassy_crust_mantle = 1._custom_real / b_rmassy_crust_mantle
317  endif
318  endif
319  rmassz_crust_mantle = 1._custom_real / rmassz_crust_mantle
320  ! inner core
321  if (rotation_val .and. exact_mass_matrix_for_rotation_val) then
322  rmassx_inner_core = 1._custom_real / rmassx_inner_core
323  rmassy_inner_core = 1._custom_real / rmassy_inner_core
324  if (simulation_type == 3) then
325  b_rmassx_inner_core = 1._custom_real / b_rmassx_inner_core
326  b_rmassy_inner_core = 1._custom_real / b_rmassy_inner_core
327  endif
328  endif
329  rmassz_inner_core = 1._custom_real / rmassz_inner_core
330  ! outer core
331  rmass_outer_core = 1._custom_real / rmass_outer_core
332 
333  ! synchronizes processes
334  call synchronize_all()
335 
336  end subroutine prepare_timerun_mass_matrices

463  subroutine prepare_timerun_convert_coord()
464 
465 ! converts x/y/z into r/theta/phi spherical coordinates
466 
467  use specfem_par
468  use specfem_par_crustmantle
469  use specfem_par_innercore
470  use specfem_par_outercore
471  implicit none
472 
473  ! local parameters
474  integer :: i,ier
475  real(kind=CUSTOM_REAL) :: rval,thetaval,phival
476 
477  ! change x, y, z to r, theta and phi once and for all
478  !
479  ! note: we merged all 3 components into a single array rstore(:,:).
480  ! this makes it more suitable for performance improvements by compilers (better prefetch, more efficient memory access)
481 
482  ! convert in the crust and mantle
483  allocate(rstore_crust_mantle(ndim,nglob_crust_mantle),stat=ier)
484  if (ier /= 0) stop 'Error allocating rstore for crust/mantle'
485 
486  do i = 1,nglob_crust_mantle
487  call xyz_2_rthetaphi(xstore_crust_mantle(i),ystore_crust_mantle(i),zstore_crust_mantle(i),rval,thetaval,phival)
488  rstore_crust_mantle(1,i) = rval
489  rstore_crust_mantle(2,i) = thetaval
490  rstore_crust_mantle(3,i) = phival
491  enddo
492 
493  ! convert in the outer core
494  allocate(rstore_outer_core(ndim,nglob_outer_core),stat=ier)
495  if (ier /= 0) stop 'Error allocating rstore for outer core'
496 
497  do i = 1,nglob_outer_core
498  call xyz_2_rthetaphi(xstore_outer_core(i),ystore_outer_core(i),zstore_outer_core(i),rval,thetaval,phival)
499  rstore_outer_core(1,i) = rval
500  rstore_outer_core(2,i) = thetaval
501  rstore_outer_core(3,i) = phival
502  enddo
503 
504  ! convert in the inner core
505  allocate(rstore_inner_core(ndim,nglob_inner_core),stat=ier)
506  if (ier /= 0) stop 'Error allocating rstore for inner core'
507 
508  do i = 1,nglob_inner_core
509  call xyz_2_rthetaphi(xstore_inner_core(i),ystore_inner_core(i),zstore_inner_core(i),rval,thetaval,phival)
510  rstore_inner_core(1,i) = rval
511  rstore_inner_core(2,i) = thetaval
512  rstore_inner_core(3,i) = phival
513  enddo
514 
515  ! old x/y/z array not needed anymore
516  deallocate(xstore_crust_mantle,ystore_crust_mantle,zstore_crust_mantle)
517  deallocate(xstore_outer_core,ystore_outer_core,zstore_outer_core)
518  deallocate(xstore_inner_core,ystore_inner_core,zstore_inner_core)
519 
520  end subroutine prepare_timerun_convert_coord

772 ! precomputes gravity factors

787  endif
788 
789  ! store g, rho and dg/dr=dg using normalized radius in lookup table every 100 m
790  ! get density and velocity from PREM model using dummy doubling flag
791  ! this assumes that the gravity perturbations are small and smooth
792  ! and that we can neglect the 3D model and use PREM every 100 m in all cases
793  ! this is probably a rather reasonable assumption
794  if (gravity_val) then
795  call make_gravity(nspl_gravity,rspl_gravity,gspl,gspl2,one_crust)

801  idoubling = 0
802  call model_prem_iso(myrank,radius,rho,drhodr,vp,vs,qkappa,qmu,idoubling,.false., &
803  one_crust,.false.,ricb,rcmb,rtopddoubleprime, &
804  r600,r670,r220,r771,r400,r80,rmoho,rmiddle_crust,rocean)
805 
806  dg = 4.0d0*rho - 2.0d0*g/radius
807 
808  minus_gravity_table(int_radius) = - g
809  minus_deriv_gravity_table(int_radius) = - dg
810  density_table(int_radius) = rho
811  minus_rho_g_over_kappa_fluid(int_radius) = - g / vp**2
812  enddo
813

821  minus_rho_g_over_kappa_fluid(int_radius) = minus_rho_g_over_kappa_fluid(nint((ricb/1000.d0 + 3.d0)*10.d0))

828  radius = ricb / r_earth

830 
831  ! distinguish between single and double precision for reals
832  minus_g_cmb = real(- g_cmb_dble, kind=custom_real)
833  minus_g_icb = real(- g_icb_dble, kind=custom_real)
834 
835  else
836 
837  ! tabulate d ln(rho)/dr needed for the no gravity fluid potential
838  do int_radius = 1,nrad_gravity
839  radius = dble(int_radius) / (r_earth_km * 10.d0)
840  idoubling = 0
841  call model_prem_iso(myrank,radius,rho,drhodr,vp,vs,qkappa,qmu,idoubling,.false., &
842  one_crust,.false.,ricb,rcmb,rtopddoubleprime, &
843  r600,r670,r220,r771,r400,r80,rmoho,rmiddle_crust,rocean)
844 
845  d_ln_density_dr_table(int_radius) = drhodr/rho
846 
847  enddo
848 
849  endif
850 
851  ! synchronizes processes
852  call synchronize_all()
853 
854  end subroutine prepare_timerun_gravity

961  c23store_crust_mantle(i,j,k,ispec) = c23store_crust_mantle(i,j,k,ispec) &
962  - two_thirds * scale_factor_minus_one * mul
963  c33store_crust_mantle(i,j,k,ispec) = c33store_crust_mantle(i,j,k,ispec) &

965  c44store_crust_mantle(i,j,k,ispec) = c44store_crust_mantle(i,j,k,ispec) &
966  + scale_factor_minus_one * mul

983  endif

986  enddo
987  enddo
988  enddo ! enddo CRUST MANTLE

991

999  else
1000  scale_factor = factor_scale_inner_core(1,1,1,ispec)
1001  endif
1002 
1003  if (anisotropic_inner_core_val) then
1004  scale_factor_minus_one = scale_factor - 1.d0
1005 
1006  mul = c44store_inner_core(i,j,k,ispec)
1007  c11store_inner_core(i,j,k,ispec) = c11store_inner_core(i,j,k,ispec) &

1020
