28 ! get backazimuth baz from event and station coordinates the, phe, ths and phs
29  subroutine get_backazimuth(the,phe,ths,phs,baz)
30

32 
33  implicit none
34 
35  double precision the, phe
36  double precision ths, phs
37  double precision az,baz,xdeg
38 
39  double precision a, a1, b, b1, c, c1
40  double precision d, d1, e, e1
41  double precision ec2, f, f1, g, g1, h, h1, onemec2, pherad
42  double precision phsrad, sc, sd, ss
43  double precision temp, therad, thg, thsrad
44 
45  double precision, parameter :: rad = 6378.160d0
46  double precision, parameter :: fl = 0.00335293d0
47  double precision, parameter :: twopideg = 360.d0
48  double precision, parameter :: c00 = 1.d0
49  double precision, parameter :: c01 = 0.25d0
50  double precision, parameter :: c02 = -4.6875d-02
51  double precision, parameter :: c03 = 1.953125d-02
52  double precision, parameter :: c21 = -0.125d0
53  double precision, parameter :: c22 = 3.125d-02
54  double precision, parameter :: c23 = -1.46484375d-02
55  double precision, parameter :: c42 = -3.90625d-03
56  double precision, parameter :: c43 = 2.9296875d-03
57  double precision, parameter :: degtokm = 111.3199d0

61  !=====================================================================
62  ! PURPOSE: To compute the distance and azimuth between locations.
63  !=====================================================================
64  ! INPUT ARGUMENTS:
65  ! THE: Event latitude in decimal degrees, North positive. [r]
66  ! PHE: Event longitude, East positive. [r]
67  ! THS: Array of station latitudes. [r]
68  ! PHS: Array of station longitudes. [r]
69  ! NS: Length of THS and PHS. [i]
70  !=====================================================================
71  ! OUTPUT ARGUMENTS:
72  ! DIST: Array of epicentral distances in km. [r]
73  ! AZ: Array of azimuths in degrees. [r]
74  ! BAZ: Array of back azimuths. [r]
75  ! XDEG: Array of great circle arc lengths. [r]
76  ! NERR: Error flag:
77  ! = 0 No error.
78  ! = 0904 Calculation failed internal consistency checks.
79  !=====================================================================
80  ! MODULE/LEVEL: DFM/4
81  !=====================================================================
82  ! GLOBAL INPUT:
83  ! MACH:
84  !=====================================================================
85  ! subroutines called:
86  ! SACLIB: SETMSG, APCMSG
87  !=====================================================================
88  ! LOCAL VARIABLES:
89  !=====================================================================
90  ! KNOWN ERRORS:
91  ! - Problem with equation for distance. See discussion below.
92  !=====================================================================
93  ! PROCEDURE:
94  ! - Calculations are based upon the reference spheroid of 1968 and
95  ! are defined by the major radius (RAD) and the flattening (FL).
96  ! - Initialize.
97  !nerr = 0
98 
99  ec2 = 2.*fl - fl*fl
100  onemec2 = 1. - ec2
101 
102  ! - Convert event location to radians.
103  ! (Equations are unstable for latitudes of exactly 0 degrees.)
104 
105  temp = the
106  if (temp == 0. ) temp = 1.0d-08
107  therad = torad*temp
108  pherad = torad*phe
109 
110  ! - Must convert from geographic to geocentric coordinates in order
111  ! to use the spherical trig equations. This requires a latitude
112  ! correction given by: 1-EC2=1-2*FL+FL*FL
113 
114  if (the == 90 .or. the == -90) then ! special attention at the poles
115  thg = the*torad ! ... to avoid division by zero.
116  else
117  thg = atan( onemec2*tan( therad ) )
118  endif
119 
120  d = sin( pherad )
121  e = -cos( pherad )
122  f = -cos( thg )
123  c = sin( thg )
124  a = f*e
125  b = -f*d
126  g = -c*e
127  h = c*d
128 
129 
130  ! -- Convert to radians.
131  temp = ths
132  if (temp == 0. ) temp = 1.0d-08
133  thsrad = torad*temp
134  phsrad = torad*phs
135 
136  ! -- Calculate some trig constants.
137  if (ths == 90 .or. ths == -90) then
138  thg = ths * torad
139  else
140  thg = atan( onemec2*tan( thsrad ) )
141  endif
142 
143  d1 = sin( phsrad )
144  e1 = -cos( phsrad )
145  f1 = -cos( thg )
146  c1 = sin( thg )
147  a1 = f1*e1
148  b1 = -f1*d1
149  g1 = -c1*e1
150  h1 = c1*d1
151  sc = a*a1 + b*b1 + c*c1
152 
153  ! - Spherical trig relationships used to compute angles.
154 
155  sd = 0.5*sqrt( ((a - a1)**2 + (b - b1)**2 + (c - &
156  c1)**2)*((a + a1)**2 + (b + b1)**2 + (c + c1)**2) )
157  xdeg = atan2( sd, sc )*todeg
158  if (xdeg < 0. ) &
159  xdeg = xdeg + twopideg
160 
161  ss = (a1 - d)**2 + (b1 - e)**2 + (c1)**2 - 2.
162  sc = (a1 - g)**2 + (b1 - h)**2 + (c1 - f)**2 - 2.
163  az = atan2( ss, sc )*todeg
164  if (az < 0. ) &
165  az = az + twopideg
166 
167  ss = (a - d1)**2 + (b - e1)**2 + (c)**2 - 2.
168  sc = (a - g1)**2 + (b - h1)**2 + (c - f1)**2 - 2.
169  baz = atan2( ss, sc )*todeg
170  if (baz < 0. ) &
171  baz = baz + twopideg
172 
173 end subroutine get_backazimuth
