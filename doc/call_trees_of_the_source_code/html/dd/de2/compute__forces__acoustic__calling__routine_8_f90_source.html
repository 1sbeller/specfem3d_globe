30 ! acoustic domains for forward or adjoint simulations (SIMULATION_TYPE == 1 or 2 )

32  use specfem_par

34  ibool_crust_mantle,ibelm_bottom_crust_mantle
35  use specfem_par_innercore, only: displ_inner_core, &

41  ! local parameters
42  real(kind=CUSTOM_REAL) :: timeval
43  ! non blocking MPI
44  ! iphase: iphase = 1 is for computing outer elements in the outer_core,
45  ! iphase = 2 is for computing inner elements in the outer core (former icall parameter)
46  integer :: iphase
47 
48  ! compute internal forces in the fluid region
49 
50  ! current simulated time
51  if (use_lddrk) then
52  timeval = real((dble(it-1)*dt+dble(c_lddrk(istage))*dt-t0)*scale_t_inv, kind=custom_real)
53  else
54  timeval = real((dble(it-1)*dt-t0)*scale_t_inv, kind=custom_real)

57  ! ****************************************************
58  ! big loop over all spectral elements in the fluid
59  ! ****************************************************
60 
61  ! distinguishes two runs: for elements on MPI interfaces, and elements within the partitions
62  do iphase = 1,2
63 
64  ! first, iphase == 1 for points on MPI interfaces (thus outer elements)
65  ! second, iphase == 2 for points purely inside partition (thus inner elements)
66  !
67  ! compute all the outer elements first, then sends out non blocking MPI communication
68  ! and continues computing inner elements (overlapping)
69 
70  if (.not. gpu_mode) then
71  ! on CPU
72  call compute_forces_outer_core(timeval,deltat,two_omega_earth, &
73  nspec_outer_core_rotation,nglob_outer_core, &
74  a_array_rotation,b_array_rotation, &
75  a_array_rotation_lddrk,b_array_rotation_lddrk, &
76  displ_outer_core,accel_outer_core, &
77  div_displ_outer_core,iphase)
78  else
79  ! on GPU
80  ! includes FORWARD_OR_ADJOINT == 1
81  call compute_forces_outer_core_gpu(mesh_pointer,iphase,timeval,1)
82 
83  ! initiates asynchronous MPI transfer
84  if (gpu_async_copy .and. iphase == 2) then
85  ! crust/mantle region
86  ! wait for asynchronous copy to finish
87  call sync_copy_from_device(mesh_pointer,iphase,buffer_send_scalar_outer_core,iregion_outer_core,1)
88  ! sends MPI buffers
89  call assemble_mpi_scalar_send_gpu(nproctot_val, &
90  buffer_send_scalar_outer_core,buffer_recv_scalar_outer_core, &
91  num_interfaces_outer_core,max_nibool_interfaces_oc, &
92  nibool_interfaces_outer_core, &
93  my_neighbours_outer_core, &
94  request_send_scalar_oc,request_recv_scalar_oc)
95  endif
96  endif
97 
98 
99  ! computes additional contributions to acceleration field
100  if (iphase == 1) then
101 
102  ! Stacey absorbing boundaries
103  if (nchunks_val /= 6 .and. absorbing_conditions) call compute_stacey_oc_forward()
104 
105  ! ****************************************************
106  ! ********** add matching with solid part **********
107  ! ****************************************************
108  ! only for elements in first matching layer in the fluid
109  if (.not. gpu_mode) then
110  ! on CPU
111  !---
112  !--- couple with mantle at the top of the outer core
113  !---
114  if (actually_couple_fluid_cmb) &
115  call compute_coupling_fluid_cmb(nglob_crust_mantle,displ_crust_mantle, &
116  ibool_crust_mantle,ibelm_bottom_crust_mantle, &
117  nglob_outer_core,accel_outer_core, &
118  normal_top_outer_core,jacobian2d_top_outer_core, &
119  wgllwgll_xy,ibool_outer_core,ibelm_top_outer_core, &
120  nspec2d_top(iregion_outer_core))
121 
122  !---
123  !--- couple with inner core at the bottom of the outer core
124  !---
125  if (actually_couple_fluid_icb) &
126  call compute_coupling_fluid_icb(nglob_inner_core,displ_inner_core, &
127  ibool_inner_core,ibelm_top_inner_core, &
128  nglob_outer_core,accel_outer_core, &
129  normal_bottom_outer_core,jacobian2d_bottom_outer_core, &
130  wgllwgll_xy,ibool_outer_core,ibelm_bottom_outer_core, &
131  nspec2d_bottom(iregion_outer_core))
132 
133  else
134  ! on GPU
135  !---
136  !--- couple with mantle at the top of the outer core
137  !---
138  if (actually_couple_fluid_cmb ) &
139  call compute_coupling_fluid_cmb_gpu(mesh_pointer,1)
140  !---
141  !--- couple with inner core at the bottom of the outer core
142  !---
143  if (actually_couple_fluid_icb ) &
144  call compute_coupling_fluid_icb_gpu(mesh_pointer,1)
145 
146  endif
147  endif ! iphase == 1
148 
149  ! assemble all the contributions between slices using MPI
150  ! in outer core
151  if (iphase == 1) then
152  ! sends out MPI interface data (non-blocking)
153  if (.not. gpu_mode) then
154  ! on CPU
155  call assemble_mpi_scalar_s(nproctot_val,nglob_outer_core, &
156  accel_outer_core, &
157  buffer_send_scalar_outer_core,buffer_recv_scalar_outer_core, &
158  num_interfaces_outer_core,max_nibool_interfaces_oc, &
159  nibool_interfaces_outer_core,ibool_interfaces_outer_core, &
160  my_neighbours_outer_core, &
161  request_send_scalar_oc,request_recv_scalar_oc)
162  else
163  ! on GPU
164  ! sends accel values to corresponding MPI interface neighbors
165 
166  ! preparation of the contribution between partitions using MPI
167  ! transfers MPI buffers to CPU
168  ! note: for asynchronous transfers, this transfers boundary region to host asynchronously. The
169  ! MPI-send is done after compute_forces_outer_core_gpu,
170  ! once the inner element kernels are launched, and the memcpy has finished.
171  call transfer_boun_pot_from_device(mesh_pointer,buffer_send_scalar_outer_core,1)
172 
173  if (.not. gpu_async_copy) then
174  ! for synchronous transfers, sending over MPI can directly proceed
175  ! outer core
176  call assemble_mpi_scalar_send_gpu(nproctot_val, &
177  buffer_send_scalar_outer_core,buffer_recv_scalar_outer_core, &
178  num_interfaces_outer_core,max_nibool_interfaces_oc, &
179  nibool_interfaces_outer_core, &
180  my_neighbours_outer_core, &
181  request_send_scalar_oc,request_recv_scalar_oc)
182  endif
183  endif
184  else
185  ! make sure the last communications are finished and processed
186  ! waits for send/receive requests to be completed and assembles values
187  if (.not. gpu_mode) then
188  ! on CPU
189  call assemble_mpi_scalar_w(nproctot_val,nglob_outer_core, &
190  accel_outer_core, &
191  buffer_recv_scalar_outer_core,num_interfaces_outer_core, &
192  max_nibool_interfaces_oc, &
193  nibool_interfaces_outer_core,ibool_interfaces_outer_core, &
194  request_send_scalar_oc,request_recv_scalar_oc)
195  else
196  ! on GPU
197  if (gpu_async_copy) then
198  ! while inner elements compute "Kernel_2", we wait for MPI to
199  ! finish and transfer the boundary terms to the device asynchronously
200  !
201  ! transfers MPI buffers onto GPU
202  call transfer_boundarypot_to_device(mesh_pointer,nproctot_val,buffer_recv_scalar_outer_core, &
203  num_interfaces_outer_core,max_nibool_interfaces_oc, &
204  request_recv_scalar_oc, &
205  iregion_outer_core,1)
206  endif
207 
208  ! waits for MPI send/receive requests to be completed and assembles values
209  call assemble_mpi_scalar_write_gpu(mesh_pointer,nproctot_val, &
210  buffer_recv_scalar_outer_core, &
211  num_interfaces_outer_core,max_nibool_interfaces_oc, &
212  request_send_scalar_oc,request_recv_scalar_oc, &
213  1) ! -- 1 == fwd accel
214  endif
215  endif ! iphase == 1
216 
217  enddo ! iphase
218 
219  ! multiply by the inverse of the mass matrix
220 
221  if (.not. gpu_mode) then
222  ! on CPU
223  call multiply_accel_acoustic(nglob_outer_core,accel_outer_core,rmass_outer_core)
224  else
225  ! on GPU
226  ! includes FORWARD_OR_ADJOINT == 1
227  call multiply_accel_acoustic_gpu(mesh_pointer,1)
228  endif
229 
230 
231  ! time schemes
232  if (use_lddrk) then
233  ! Runge-Kutta scheme
234  call update_veloc_acoustic_lddrk()
235  else
236  ! Newmark time scheme
237  call update_veloc_acoustic_newmark()
238  endif
239 
240  end subroutine compute_forces_acoustic
